\newcommand{\do@tikzline@numberalign}{%
  \newbox\@temp%
  \setbox\@temp=\hbox{X}%
  \hspace{-1\wd\@temp}%
  \hspace{-3\paperwidth}\unhbox\@temp\hspace{3\paperwidth}%
}
\newcommand{\do@tikzline}[1]{%
  \do@tikzline@numberalign%
  \getline@num%
  \edlabel{#1}%
  \l@dld@ta%
  \new@line%
  \hfill%
  \l@drd@ta%
}
