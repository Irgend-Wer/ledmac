\begin{document}

  \begin{tikzpicture}[sibling distance=5cm,
                      edge from parent/.style={draw,thick,-}]
   \node(\thetikzline){Prova\stepcounter{tikzline}}
        child {node[align=left] {Second line - left } }
        child {node(\thetikzline)[align=right] {Second line - right\stepcounter{tikzline}}
            child {node (\thetikzline)[align=left] { Third line - left\stepcounter{tikzline}}}
            child {node {Third line }}
        };


  % loop over the "left" nodes
  \foreach \Node in \lines {%\y1=y-coord of the node
    \path let \p1=($ (\Node) $) in node at (-5,\y1){\Node};
  }

  % loop over the "right" nodes
  \foreach \Node in \lines {%\y1=y-coord of the node
    \path let \p1=($ (\Node) $) in node at (7,\y1){\Node};
  }

\end{tikzpicture}
