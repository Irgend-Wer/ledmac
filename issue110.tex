\newcommand{\@sameword}[1]{%
  \ifnumgreater{\csuse{sw@\the\absline@num @#1}}{1}%
    {\expandafter\showwordrank{\expandafter#1}{\csuse{the@sw}}}%
    {#1}%
}

\newcommand{\sameword}[1]{%
	#1%
	\write\linenum@out{\string\@@sw{#1}}%
	\csnumgdef{sw@#1}{\csuse{sw@#1}+1}%
	\xdef\the@sw{\csuse{sw@#1}}%
}
\def\@@sw#1{
  \csnumgdef
   {sw@\the\absline@num @#1}
   {\csuse{sw@\the\absline@num @#1}+1}
}
