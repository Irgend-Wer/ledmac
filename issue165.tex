  \fi%

    \led@err@PstartNotNumbered%
    \beginnumbering%
  \fi%

    \led@err@PstartInPstart%
    \pend%
  \fi%

  \global\let\next@insert=\empty%
  \begingroup\normal@pars%
  \global\setbox\raw@text=\vbox\bgroup%

       \fi%
      \fi%
     \fi%
  \numberedpar@true%
  \iflabelpstart%
    \protected@edef\@currentlabel%
       {\p@pstart\thepstart}
  \fi%
}
