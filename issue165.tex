\def\eled@chapter#1#2#3#4#5{%
    \ifstrempty{#4}%
      {%
      \ifstrempty{#1}%
	    {\global\csdef{eled@sectionning@#3#5}{\let\edtext=\dummy@edtext\chapter{#2}}}
	    {\global\csdef{eled@sectionning@#3#5}{\let\edtext=\dummy@edtext\chapter[#1]{#2}}}
	  }%
	  {%
	  \ifstrempty{#1}%
	    {\global\csdef{eled@sectionning@#3#5}{\let\edtext=\dummy@edtext\chapter*{#2}}}
	    {\global\csdef{eled@sectionning@#3#5}{\let\edtext=\dummy@edtext\chapter*[#1]{#2}}}%Bug in LaTeX!
	  }%
	\listcsgadd{eled@sections@@#5}{#3}%
	}

\renewcommand*{\pstart}{%
  \global\advance\l@dnumpstartsL \@ne
  \if@nobreak%
    \let\@oldnobreak\@nobreaktrue%
  \else%
    \let\@oldnobreak\@nobreakfalse%
  \fi%
  \@nobreaktrue%
  \ifnumbering \else%
    \led@err@PstartNotNumbered%
    \beginnumbering%
  \fi%
  \ifnumberedpar@%
    \led@err@PstartInPstart%
    \pend%
  \fi%
  \list@clear{\inserts@list}%
  \global\let\next@insert=\empty%
  \begingroup\normal@pars%
  \global\setbox\raw@text=\vbox\bgroup%
    \ifautopar\else%
    \ifnumberpstart%
      \ifinstanza\else%
       \ifsidepstartnum\else%
         \thepstart%
        \fi%
       \fi%
      \fi%
     \fi%
  \numberedpar@true%
  \iflabelpstart%
    \protected@edef\@currentlabel%
       {\p@pstart\thepstart}
  \fi%
}

\def\endnumberingR{%
  \ifnumberingR
    \global\numberingRfalse
    \normal@pars
    \ifl@dpairing
      \global\pst@rtedRfalse
    \else
      \ifx\insertlines@listR\empty\else
        \global\noteschanged@true
      \fi
      \ifx\line@listR\empty\else
        \global\noteschanged@true
      \fi
    \fi
    \ifnoteschanged@
      \led@mess@NotesChanged
    \fi
  \else
    \led@err@NumberingNotStarted
  \fi
  \endgroup
  \immediate\closeout\eled@sectionningR@out
  }

\def\endnumbering{%
  \ifnumbering
    \global\numberingfalse
    \normal@pars
    \ifl@dpairing
      \global\pst@rtedLfalse
    \else
      \ifx\insertlines@list\empty\else
        \global\noteschanged@true
      \fi
      \ifx\line@list\empty\else
        \global\noteschanged@true
      \fi
    \fi
    \ifnoteschanged@
      \led@mess@NotesChanged
    \fi
  \else
    \led@err@NumberingNotStarted
  \fi
  \autoparfalse
  \immediate\closeout\eled@sectionning@out
  \ifl@dpairing\else
    \endgroup
  \fi
}
