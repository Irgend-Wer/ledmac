Quand il y a une commande similaire dans le texte de gauche, aucun problème. Le MWE fonctionne comme écrit; mais si nous supprimons le \ edtext et son deuxième argument du texte de gauche, nous obtenons le problème démontré dans le PNG.
\documentclass{memoir}

\usepackage{reledmac}
	\Xendtxtbeforenotes{\vskip 6pt\footnoterule}

\usepackage{reledpar} 
	\setstanzaindents{1,0}
	\setcounter{stanzaindentsrepetition}{1}

\begin{document}

\begin{pairs}
    \begin{Leftside}
        \beginnumbering
            \begin{astanza}
                %\edtext{Left}{\Aendnote{<left endnote>}} % remove this edtext and endnote
                paragraph\&
            \end{astanza}
        \endnumbering
    \end{Leftside}
    \begin{Rightside}
        \beginnumbering
        \autopar
        And \edtext{right}{\Aendnote{<right endnote>}} paragraph.\pend
        \endnumbering
    \end{Rightside}

\end{pairs} 
\Columns
\doendnotesbysection{A}

\end{document}
