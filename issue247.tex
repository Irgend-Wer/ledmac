\newcommand{\unvxh@l}[1]{%
  \vbadness=10000\setbox0=\vbox{\unvbox#1%
    \global\setbox1=\lastbox}%
  \unhbox1%
  \unskip%            % remove \rightskip,
  \unskip%            % remove \parfillskip,
  \unpenalty%         % remove \penalty of 10000,
  }

\def\continue@longnote{%
  \def\do##1{%
    \ifvoid\csuse{##1toolong@para}\else%
      \setbox0=\box\csuse{##1toolong@para}%
      \setbox2=\vbox{}%
      \vbadness=10000%
      \splittopskip=0pt%
      \csuse{Xnotefontsize@##1}%
      \@whiledim\ht0>0pt\do{%
        \vsplit0 to \baselineskip%
        \setbox1=\lastbox%
        \ifdim\ht0>0pt%
          \setbox1=\hbox{\unvxh@l{1}}%
        \else%
          \setbox1=\hbox{\unvxh1[##1]}%
        \fi%
      \setbox2=\vbox{\unvbox2\box1}%
  }%
  \global\csuse{##1longpara@total}=\dimexpr\ht2+\dp2\relax%
  \insert\csuse{##1footins}{\unvbox2}%
  \ifvoid\csuse{##1keept@para}%
  \else%
    \global\csuse{##1longpara@total}=\dimexpr%
      \csuse{##1longpara@total}%
      +\ht\csuse{##1keept@para}%
      +\dp\csuse{##1keept@para}%
    \relax%
    \insert\csuse{##1footins}{\unvbox\csuse{##1keept@para}}%
  \fi%      
    \fi%
 }%
 \dolistloop{\@series}%
}
